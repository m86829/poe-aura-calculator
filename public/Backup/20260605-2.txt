        function bindMiltonPowerControls(element, defaultValue) {
            element.addEventListener("wheel", function (e) {
                if (document.activeElement !== element) return;
                e.preventDefault();
                if (element.tagName === "SELECT") {
                    let idx = element.selectedIndex;
                    if (e.deltaY < 0 && idx > 0) element.selectedIndex = idx - 1;
                    if (e.deltaY > 0 && idx < element.options.length - 1) element.selectedIndex = idx + 1;
                } else if (element.type === "number") {
                    let val = parseFloat(element.value) || 0;
                    element.value = e.deltaY < 0 ? val + 1 : val - 1;
                }
                element.dispatchEvent(new Event("input"));
                element.dispatchEvent(new Event("change"));
            }, { passive: false });

            element.addEventListener("mousedown", function (e) {
                if (e.button === 1) {
                    e.preventDefault();
                    resetField(element, defaultValue);
                    // 🌟 修正：如果是下拉選單被重設，強制觸發更新事件，讓右手邊說明與數值一起洗白歸 0
                    if (element.tagName === "SELECT") {
                        element.dispatchEvent(new Event("change"));
                    }
                }
            });

            element.addEventListener("keydown", function (e) {
                if (["Delete", "Backspace", "Escape"].includes(e.key)) {
                    e.preventDefault();
                    resetField(element, defaultValue);
                    // 🌟 修正：如果是下拉選單被重設，強制觸發更新事件，讓右手邊說明與數值一起洗白歸 0
                    if (element.tagName === "SELECT") {
                        element.dispatchEvent(new Event("change"));
                    }
                }
            });
        }

        function resetField(element, defaultValue) {
            if (element.type === "checkbox") element.checked = defaultValue;
            else element.value = defaultValue;
            element.dispatchEvent(new Event("input"));
            element.dispatchEvent(new Event("change"));
            saveState(); calculateAura();
        }

        function calculateAura() {
            let treeMRes = 0, treeLRes = 0, treeEff = 0;
            TREE_CONFIG.forEach(t => {
                let cEff = 0, cM = 0, cL = 0;

                if (t.isSpecial) {
                    // 🌟 特殊節點：偵測 A 勾選狀態
                    if (document.getElementById(`tree-special-chk-${t.id}`)?.checked) {
                        cEff = t.eff; cM = t.mres; cL = t.lres;
                    }
                } else {
                    // 🌟 前六個星盤：精確加總 A, B, C, D 字母對應的個別加成屬性
                    t.items.forEach(item => {
                        if (document.getElementById(`node-chk-${t.id}-${item.node}`)?.checked) {
                            cEff += item.eff;
                            cM += item.mres;
                            cL += item.lres;
                        }
                    });
                }

                const targetEff = document.getElementById(`tree-summary-eff-${t.id}`);
                const targetM = document.getElementById(`tree-summary-mres-${t.id}`);
                const targetL = document.getElementById(`tree-summary-lres-${t.id}`);

                // ==========================================================================
                // 🌟 單格獨立變色判定：只有數值大於 0 的獨立儲存格才會觸發彩色放大亮燈
                // ==========================================================================

                // 1. 光環效果獨立亮燈 (紫色加成)
                if (targetEff) {
                    if (cEff > 0) {
                        targetEff.innerText = `+${cEff}%`;
                        targetEff.style.backgroundColor = "#eddcf2"; // 高亮紫色底
                        targetEff.style.color = "#6B21A8";          // 深紫色字
                        targetEff.style.fontSize = "1.05rem";        // 放大字體
                    } else {
                        targetEff.innerText = "0%";
                        targetEff.style.backgroundColor = "#ffffff"; // 預設乾淨白底
                        targetEff.style.color = "#666666";          // 預設內斂灰字
                        targetEff.style.fontSize = "0.82rem";        // 恢復正常大小
                    }
                }

                // 2. 魔力保留效用獨立亮燈 (藍色加成)
                if (targetM) {
                    if (cM > 0) {
                        targetM.innerText = `${cM}%`;
                        targetM.style.backgroundColor = "#ddebf7";  // 高亮藍色底
                        targetM.style.color = "#2563eb";            // 深藍色字
                        targetM.style.fontSize = "1.05rem";        // 放大字體
                    } else {
                        targetM.innerText = "0%";
                        targetM.style.backgroundColor = "#ffffff"; // 預設乾淨白底
                        targetM.style.color = "#666666";          // 預設內斂灰字
                        targetM.style.fontSize = "0.82rem";        // 恢復正常大小
                    }
                }

                // 3. 生命保留效用獨立亮燈 (橘色加成)
                if (targetL) {
                    if (cL > 0) {
                        targetL.innerText = `${cL}%`;
                        targetL.style.backgroundColor = "#fce4d6";  // 高亮橘色底
                        targetL.style.color = "#c2410c";            // 深橘色字
                        targetL.style.fontSize = "1.05rem";        // 放大字體
                    } else {
                        targetL.innerText = "0%";
                        targetL.style.backgroundColor = "#ffffff"; // 預設乾淨白底
                        targetL.style.color = "#666666";          // 預設內斂灰字
                        targetL.style.fontSize = "0.82rem";        // 恢復正常大小
                    }
                }

                treeMRes += cM; treeLRes += cL; treeEff += cEff;
            });

            let gearMRes = 0, gearLRes = 0, gearEff = 0;
            Object.keys(GEAR_SLOT_OPTIONS).forEach((slotKey, idx) => {
                let m = parseFloat(document.getElementById(`gear-mres-${idx}`)?.value) || 0;
                let l = parseFloat(document.getElementById(`gear-lres-${idx}`)?.value) || 0;
                let e = parseFloat(document.getElementById(`gear-eff-${idx}`)?.value) || 0;
                let v = parseFloat(document.getElementById(`gear-val-${idx}`)?.value) || 0;

                if (slotKey.includes("珠寶")) {
                    gearMRes += (m * v); gearLRes += (l * v); gearEff += (e * v);
                } else {
                    gearMRes += m; gearLRes += l; gearEff += e;
                }
            });

            document.getElementById("calc-total-life-eff").innerText = treeLRes + gearLRes;
            document.getElementById("calc-total-mana-eff").innerText = treeMRes + gearMRes;
            document.getElementById('summary-effect-calc').innerText = `+${treeEff + gearEff}%`;

            let finalLifeEff = treeLRes + gearLRes;
            let finalManaEff = treeMRes + gearMRes;
            if (document.getElementById("override-global-enable").checked) {
                finalLifeEff = parseFloat(document.getElementById("override-life-input").value) || 0;
                finalManaEff = parseFloat(document.getElementById("override-mana-input").value) || 0;
            }

            const baseLife = Math.floor(parseFloat(document.getElementById('base-life').value)) || 0;
            const baseMana = Math.floor(parseFloat(document.getElementById('base-mana').value)) || 0;

            let sumPctLife = 0, sumPctMana = 0;
            let flatLifeSum = 0, flatManaSum = 0;

            document.querySelectorAll('.pct-row').forEach(row => {
                const id = row.id.replace('row-', '');
                const isChecked = document.getElementById(`chk-${id}`).checked;
                const dispLife = document.getElementById(`cost-life-${id}`);
                const dispMana = document.getElementById(`cost-mana-${id}`);

                if (isChecked) {
                    let basePct = parseFloat(row.getAttribute('data-base')) || 0;
                    const slotEff = parseFloat(document.getElementById(`eff-${id}`).value) || 0;
                    const enlLv = parseInt(document.getElementById(`enl-${id}`).value) || 0;
                    const gemMult = parseFloat(document.getElementById(`mult-${id}`).value) || 100;
                    const arrLv = parseInt(document.getElementById(`arr-${id}`).value) || -1;
                    const extraSupportMult = parseFloat(document.getElementById(`sup-${id}`).value) || 1.0;

                    let enlMultiplier = (enlLv > 0 && POEDB.enlighten[enlLv] !== undefined) ? POEDB.enlighten[enlLv] : 1.0;
                    let arrMultiplier = 1.0;
                    let isArroganceActive = false;

                    if (arrLv === 0) { arrMultiplier = 1.0; isArroganceActive = true; }
                    else if (arrLv > 0 && POEDB.arrogance[arrLv] !== undefined) { arrMultiplier = POEDB.arrogance[arrLv]; isArroganceActive = true; }

                    if (isArroganceActive) {
                        let finalPct = (basePct * arrMultiplier * enlMultiplier * (gemMult / 100) * extraSupportMult * 100) / (100 + finalLifeEff + slotEff);
                        finalPct = Math.floor(finalPct * 10000) / 10000; sumPctLife += finalPct;
                        dispLife.innerText = `${finalPct.toFixed(1)}%`; dispMana.innerText = "-";
                    } else {
                        let finalPct = (basePct * enlMultiplier * (gemMult / 100) * extraSupportMult * 100) / (100 + finalManaEff + slotEff);
                        finalPct = Math.floor(finalPct * 10000) / 10000; sumPctMana += finalPct;
                        dispMana.innerText = `${finalPct.toFixed(1)}%`; dispLife.innerText = "-";
                    }
                } else {
                    if (dispLife) dispLife.innerText = "-";
                    if (dispMana) dispMana.innerText = "-";
                }
            });

            document.querySelectorAll('.flat-row').forEach(row => {
                const aura = row.getAttribute('data-aura-type');
                const isChecked = document.getElementById(`chk-${aura}`).checked;
                const dispLife = document.getElementById(`cost-life-${aura}`);
                const dispMana = document.getElementById(`cost-mana-${aura}`);

                if (isChecked) {
                    const lv = parseInt(document.getElementById(`lv-${aura}`).value) || 20;
                    const slotEff = parseFloat(document.getElementById(`eff-${aura}`).value) || 0;
                    const enlLv = parseInt(document.getElementById(`enl-${aura}`).value) || 0;
                    const gemMult = parseFloat(document.getElementById(`mult-${aura}`).value) || 100;
                    const arrLv = parseInt(document.getElementById(`arr-${aura}`).value) || -1;
                    const extraSupportMult = parseFloat(document.getElementById(`sup-${aura}`).value) || 1.0;

                    const baseFlat = POEDB[aura][lv] || 0;
                    let enlMultiplier = (enlLv > 0 && POEDB.enlighten[enlLv] !== undefined) ? POEDB.enlighten[enlLv] : 1.0;
                    let arrMultiplier = (arrLv > 0 && POEDB.arrogance[arrLv] !== undefined) ? POEDB.arrogance[arrLv] : 1.0;

                    if (arrLv >= 0) {
                        let reserved = Math.floor((baseFlat * arrMultiplier * enlMultiplier * (gemMult / 100) * extraSupportMult * 100) / (100 + finalLifeEff + slotEff));
                        flatLifeSum += reserved; dispLife.innerText = `${reserved.toLocaleString()} HP`; dispMana.innerText = "-";
                    } else {
                        let reserved = Math.floor((baseFlat * enlMultiplier * (gemMult / 100) * extraSupportMult * 100) / (100 + finalManaEff + slotEff));
                        flatManaSum += reserved; dispMana.innerText = `${reserved.toLocaleString()} MP`; dispLife.innerText = "-";
                    }
                } else {
                    if (dispLife) dispLife.innerText = "-";
                    if (dispMana) dispMana.innerText = "-";
                }
            });

            let pctLifePoints = Math.floor((baseLife * sumPctLife) / 100);
            let pctManaPoints = Math.floor((baseMana * sumPctMana) / 100);
            const finalLife = baseLife - pctLifePoints - flatLifeSum;
            const finalMana = baseMana - pctManaPoints - flatManaSum;

            const elLife = document.getElementById('remaining-life');
            const elMana = document.getElementById('remaining-mana');
            elLife.innerText = finalLife.toLocaleString();
            elMana.innerText = finalMana.toLocaleString();
            elLife.className = finalLife >= 0 ? "sub-pts-value val-positive" : "sub-pts-value val-negative";
            elMana.className = finalMana >= 0 ? "sub-pts-value val-positive" : "sub-pts-value val-negative";

            document.getElementById('life-reserved-percent').innerText = `${sumPctLife.toFixed(2)}%`;
            document.getElementById('mana-reserved-percent').innerText = `${sumPctMana.toFixed(2)}%`;
        }

        function saveState() {
            const state = {};
            document.querySelectorAll('input, select').forEach(i => { if (i.id) state[i.id] = i.type === 'checkbox' ? i.checked : i.value; });
            historyStack.push(JSON.stringify(state)); redoStack = [];
        }

        function restoreSnapshot(snap) {
            Object.keys(snap).forEach(id => {
                const el = document.getElementById(id);
                if (el) el.type === 'checkbox' ? el.checked = snap[id] : el.value = snap[id];
            });
            calculateAura();
        }

        window.undo = function () { if (historyStack.length > 1) { redoStack.push(historyStack.pop()); restoreSnapshot(JSON.parse(historyStack[historyStack.length - 1])); } };
        window.redo = function () { if (redoStack.length > 0) { const next = redoStack.pop(); historyStack.push(next); restoreSnapshot(JSON.parse(next)); } };

        window.switchTab = function (tabId) {
            document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));
            document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));
            document.getElementById(tabId).classList.add('active');
            const btn = document.getElementById(`btn-${tabId}`); if (btn) btn.classList.add('active');
        };

        document.addEventListener('DOMContentLoaded', () => {
            renderCalculatorTable();
            initSubPages();
            bindMiltonPowerControls(document.getElementById("base-life"), 4000);
            bindMiltonPowerControls(document.getElementById("base-mana"), 1000);

            document.querySelectorAll('input, select').forEach(i => {
                if (i.id && !i.id.includes("base")) {
                    i.addEventListener(i.type === 'checkbox' ? 'change' : 'input', () => { saveState(); calculateAura(); });
                }
            });

            saveState(); calculateAura();
        });
    </script>
